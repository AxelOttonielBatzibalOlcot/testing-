Codigo de dos leds intermitentes en arduino

int led1=13; /*Identificacion de los componentes. */

int led2=12; /*Identificacion de los componentes. */

void setup()

{

pinMode(led1, OUTPUT); /*Verificacion si los componentes son de entrada o de salida*/

pinMode(led2, OUTPUT); /*Verificacion si los componentes son de entrada o de salida*/

}

void loop() /*Parte del loop*/

{

digitalWrite(led1, HIGH); /*La orden para que el primer led prenda*/

delay(1000); /*El primer led enciende por un segundo*/

digitalWrite(led1, LOW); /*El primer led se apaga*/

delay(1000); /*El primer led se apaga por un segundo*/

digitalWrite(led2, HIGH); /*La orden para que el segundo led prenda*/

delay(1000); /*El segundo led enciende por un segundo*/

digitalWrite(led2, LOW); /*El segundo led se apaga*/

delay(1000); /*El segundo led se apaga por un segundo*/

}
